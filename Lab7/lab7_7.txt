DATA SEGMENT PARA PUBLIC 'DATA'
DATA ENDS

CODE SEGMENT PARA PUBLIC 'CODE'
START PROC FAR
ASSUME CS:CODE, DS:DATA
PUSH DS 
XOR AX, AX
PUSH AX
MOV AX, DATA
MOV DS, AX

	MOV AH, 0
	MOV AL, 0FH
	CALL P1
	
	RET
START ENDP
P1 PROC NEAR
	CMP AL, 0Ah
	JL L1		
	ADD AL, 37h
	RET

	L1:
	ADD AL, 30H
	RET
P1 ENDP
CODE ENDS
END START
