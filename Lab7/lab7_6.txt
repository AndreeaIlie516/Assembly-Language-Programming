DATA SEGMENT PARA PUBLIC 'DATA'
DATA ENDS

CODE SEGMENT PARA PUBLIC 'CODE'
START PROC FAR
ASSUME CS:CODE, DS:DATA
PUSH DS 
XOR AX,AX
PUSH AX
MOV AX,DATA
MOV DS, AX

	MOV BL, 10H
	XOR DX, DX

	L: 
	MOV AH, 01H
	INT 21H		
	
	CMP AL, 0DH
	JE NEXT
	SUB AL, '0'
	MOV CL, AL
	XOR CH, CH
	MOV AX, DX
	MUL BL
	ADD AX, CX
	MOV DX, AX
	JMP L

	NEXT:
	RET

START ENDP
CODE ENDS
END START
