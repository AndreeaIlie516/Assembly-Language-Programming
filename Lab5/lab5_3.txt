DATA SEGMENT PARA PUBLIC 'DATA'
	REZ DB 0
	NUM1 DB 2
	NUM2 DB 4 
DATA ENDS

STCK SEGMENT PARA STACK 'STACK'
      STACK_START LABEL WORD
STCK ENDS

CODE SEGMENT PARA PUBLIC 'CODE'
START PROC FAR
   ENTRY PROC NEAR
	ASSUME CS:CODE, DS:DATA
	PUSH DS
	XOR AX,AX
	PUSH AX
	MOV AX,DATA
	MOV DS,AX
	
	MOV BL,1		
	MOV AL,3		
	MUL NUM2	; NUM2*AL		
	ADD BL,AL	; NUM2*AL+BL	
	MOV AL,4		
	MUL NUM1	; AL*NUM1		
	ADD REZ,AL		
	ADD REZ,BL	; AL*NUM1+(NUM2*AL)+BL	
	
   ENTRY ENDP
	RET
START ENDP
CODE ENDS
END START
